  5. Operaciones de variables
    • Aritméticos Ejemplo let resultado=numero1+numeros2;
    • Concatenación Ejemplo let nombreUsuario = "Xurxo"; let apellidosUsuario="González"; let concatenacion=nombreUsuario+" "+apellidosUsuario; document.write(`Su nombre completo es ${concatenacion}.`);
